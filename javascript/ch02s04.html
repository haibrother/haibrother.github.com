2.4. 获取url所有参数，赋予给指定标签


//需要先加载jquery
 $(function(){
        var thisUrl = window.location.toString();
        if(thisUrl.indexOf("?")>-1)
        {
            $(".btn_real,.btn_demo,.btn_dep,.proc_real,.proc_dep").each(function(index){
                $(this).attr("href",$(this).attr("href")+"?"+thisUrl.split("?")[1])
            });
        }
    })
