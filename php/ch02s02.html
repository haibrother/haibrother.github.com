2.2. 实践2：PHP正则匹配中文和提取中文


$preg_match = "/[\x{4e00}-\x{9fa5}]*/u";  #############utf-8编码环境提取
//$preg_match = "/([".chr(0xb0)."-".chr(0xf7)."][".chr(0xa1)."-".chr(0xfe)."])*/";  #############ANSI编码环境提供
$str = '中国a日本b美国c';
//检测是否含有中文
#var_dump(preg_match($preg_match,$str));exit;
//提取中文
preg_match_all($preg_match,$str,$match); 
//preg_match_all($preg_match,$str,$match); ###################ANSI编码环境提供
foreach($match as $k=>$v){
	foreach($v as $key=>$value){
		if(empty($value)){
			unset($match[$k][$key]);
		}
	}
}
if(!empty($match)){
	$s = implode('-',$match[0]);
	file_put_contents('test.txt',$s);
}
